#include <iostream>
#include <vector>
#include <set> 
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int missingNumber(vector<int>& v) 
    {
        int n=v.size();
        vector<int> dp(n+1, 0), ans;
        for(int i=0; i<n; i++)  dp[v[i]]++;
        for(int i=1; i<=n; i++) if(dp[i]>1) return i;
        return 0;
    }
};

int main()
{
   vector<int> v = {2,2,2,2,2}; 
   Solution s;
   cout<<s.missingNumber(v);
   return 0;
}
